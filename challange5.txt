#include<iostream>
#include<vector>
using namespace std;
int main(){
    vector<int> arr = {16,17,4,3,5,2};
    int n = arr.size();
    int k, y = 0;
    vector <int> arr2;
    for(int i = 0; i < n; i++){
        bool leader = true;
        for (int j = i+1; j < n; j++)
        {
            if(arr[i] <= arr[j]){
                leader = false;
                break;
            }
        }
        if(leader){
            arr2.push_back(arr[i]);
        }
        
    }
    for (int i = 0; i < arr2.size(); i++)
    {
        cout<<arr2[i]<<" ";
    }
    return 0;
    
}
